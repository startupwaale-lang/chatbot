export default async function handler(req, res) { if (req.method !== "POST") { return res.status(405).json({ error: "Only POST allowed" }); } const { message } = req.body; try { const apiRes = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${process.env.OPENAI_API_KEY}` }, body: JSON.stringify({ model: "gpt-3.5-turbo", messages: [{ role: "user", content: message }] }) }); const data = await apiRes.json(); return res.status(200).json({ reply: data.choices?.[0]?.message?.content || "No response" }); } catch (error) { return res.status(500).json({ error: error.message }); } }
